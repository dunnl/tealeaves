Module Type AtomModule <: UsualDecidableType.

  Parameter atom : Set.

  Definition t := atom.

  Parameter eq_dec : ∀ x y : atom, {x = y} + {x ≠ y}.

  Parameter atom_fresh_for_list :
    ∀ (xs : list t), {x : atom | ¬ x ∈ xs}.

  Parameter fresh : list atom → atom.

  Parameter fresh_not_in : ∀ l, ¬ fresh l ∈ l.

  Parameter nat_of : atom → nat.

  #[export] Hint Resolve eq_dec : core.

  Include HasUsualEq <+ UsualIsEq <+ UsualIsEqOrig.

End AtomModule.

Implementation of atoms

Module Atom : AtomModule.

  Definition atom := nat.

  Definition t := atom.

  Definition eq_dec := PeanoNat.Nat.eq_dec.

  Include HasUsualEq <+ UsualIsEq <+ UsualIsEqOrig.

  Lemma max_lt_r : ∀ x y z,
      x ≤ z → x ≤ max y z.
  Proof.
    induction x.
    - auto with arith.
    - induction y.
      + auto with arith.
      + cbn. induction z. lia. auto with arith.
  Qed.

  Lemma nat_list_max : ∀ (xs : list nat),
      { n : nat | ∀ x, List.In x xs → x ≤ n }.
  Proof.
    induction xs as [ | x xs [y H] ].
    - ∃ 0. inversion 1.
    - ∃ (max x y). intros z J.
      cbn in J. destruct J as [K | K].
      + subst. auto with arith.
      + auto using max_lt_r.
  Qed.

  Lemma atom_fresh_for_list :
    ∀ (xs : list nat), { n : nat | ¬ n ∈ xs }.
  Proof.
    intros xs. destruct (nat_list_max xs) as [x H].
    ∃ (S x). intros J. lapply (H (S x)). lia. trivial.
  Qed.

  Definition fresh (l : list atom) :=
    match atom_fresh_for_list l with
      (exist _ x _) ⇒ x
    end.

  Lemma fresh_not_in : ∀ l, ¬ fresh l ∈ l.
  Proof.
    intro l. unfold fresh.
    destruct atom_fresh_for_list. auto.
  Qed.

  Definition nat_of := fun (x : atom) ⇒ (x : nat).

End Atom.
