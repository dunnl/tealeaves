Rewriting lemmas for foldMapd

Section term_foldMapd_rewrite.

  Context {A M : Type} (f : nat × A → M) `{Monoid M}.

  Lemma term_foldMapd1 : ∀ (a : A),
      foldMapd f (tvar a) = f (Ƶ, a).
  Proof.
    intros. simplify_foldMapd. reflexivity.
  Qed.

  Lemma term_foldMapd2 : ∀ X (t : Lam A),
      foldMapd f (λ X t) = foldMapd (f ⦿ 1) t.
  Proof.
    intros. simplify_foldMapd. reflexivity.
  Qed.

  Lemma term_foldMapd3 : ∀ (t1 t2 : Lam A),
      foldMapd f (⟨t1⟩ (t2)) = foldMapd f t1 ● foldMapd f t2.
  Proof.
    intros. simplify_foldMapd. reflexivity.
  Qed.

End term_foldMapd_rewrite.

  Qed.

      (n, l) ∈d t = (S n, l) ∈d (λ X t).
  Proof.
    intros.
    simplify.
    conclude.
  Qed.

  Lemma term_ind2_nZ : ∀ (t : term) (l : LN) (n : nat) (X : typ),
      (n, l) ∈d (λ X t) → n ≠ 0.
  Proof.

    destruct n.

Rewriting lemmas for tolist, toset, ∈

Section term_container_rewrite.

  Variable
    (A : Type).

  Lemma tolist_tvar_rw1: ∀ (x: A),
      tolist (tvar x) = [x].
  Proof.
    intros.
    unfold tolist.
    unfold Tolist_Traverse.
    simplify.
    reflexivity.
  Qed.

  Lemma tolist_term_rw2: ∀ (X: typ) (t: term),
      tolist (lam X t) = tolist t.
  Proof.
    intros.
    simplify.
    conclude.
  Qed.

  Lemma tolist_term_rw3: ∀ (t1 t2: term),
      tolist (app t1 t2) = tolist t1 ++ tolist t2.
  Proof.
    intros.
    simplify.
    conclude.
  Qed.

  Lemma toset_tvar_rw1: ∀ (x: A),
      tosubset (tvar x) = {{x}}.
  Proof.
    intros.
    simplify.
    fixme.
  Qed.

  Lemma toset_term_rw2: ∀ (X: typ) (t: term),
      tosubset (lam X t) = tosubset t.
  Proof.
    intros.
    simplify.
    conclude.
  Qed.

  Lemma toset_term_rw3: ∀ (t1 t2: term),
      tosubset (app t1 t2) = tosubset t1 ∪ tosubset t2.
  Proof.
    intros.
    simplify.
    conclude.
  Qed.

  Lemma in_term_rw1: ∀ (x y: A),
      x ∈ tvar y = (x = y).
  Proof.
    intros.
    simplify.
    simpl_subset.
    conclude.
  Qed.

  Lemma in_term_rw2: ∀ (y: A) (X: typ) (t: Lam A),
      y ∈ (lam X t) = y ∈ t.
  Proof.
    intros.
    simplify.
    conclude.
  Qed.

  Lemma in_term_3: ∀ (t1 t2: Lam A) (y: A),
      y ∈ (app t1 t2) = (y ∈ t1 ∨ y ∈ t2).
  Proof.
    intros.
    simplify.
    conclude.
  Qed.

End term_container_rewrite.
