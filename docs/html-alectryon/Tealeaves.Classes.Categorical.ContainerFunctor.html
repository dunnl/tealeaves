(** ** Functor typeclass *)
(******************************************************************************)
Class ContainerFunctor
  (F : Type -> Type)
  `{Map F} `{ToSubset F} :=
  { cont_natural :> Natural (@tosubset F _);
    cont_functor :> Functor F;
    cont_pointwise : forall (A B : Type) (t : F A) (f g : A -> B),
      (forall a, a ‚àà t -> f a = g a) -> map F f t = map F g t;
  }.

(** ** [ToSubset]-preserving natural transformations *)
(******************************************************************************)
Class ContainerTransformation
  {F G : Type -> Type}
  `{Map F} `{ToSubset F}
  `{Map G} `{ToSubset G}
  (Œ∑ : F ‚áí G) :=
  { cont_trans_natural : Natural Œ∑;
    cont_trans_commute : forall A, tosubset (F := F) = tosubset (F := G) ‚àò Œ∑ A;
  }.
