Ltac simplify_extract :=
  first [ change ((?f ‚àò extract) (?dec, ?var)) with (f var)
        | change ((?g ‚àò (?f ‚àò extract)) (?dev, ?var)) with ((g ‚àò f) var)
    ].

Ltac simplify_map_ret :=
  change ((map ret ‚àò ?f) ?leaf) with (map ret (f leaf));
  rewrite map_to_ap.

Ltac simplify_ret :=
  change ((ret ‚àò ?f) ?leaf) with (ret (f leaf)).

Ltac simplify_leaf :=
  try simplify_extract;
  try simplify_map_ret;
  try simplify_ret;
  unfold_all_transparent_tcs.
