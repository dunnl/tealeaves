(** * Beck distribution laws *)
(******************************************************************************)
Class BeckDistribution (U T : Type -> Type)
  := bdist : forall {A : Type}, U (T A) -> T (U A).

Arguments bdist U T%function_scope {BeckDistribution} {A}%type_scope _.

#[local] Notation "'Œ¥'" := (bdist) : tealeaves_scope.

Section BeckDistributiveLaw.

  Context
    (U T : Type -> Type)
    `{Monad U}
    `{Monad T}
    `{BeckDistribution U T}.

  Class BeckDistributiveLaw :=
    { bdist_monad_l : Monad T;
      bdist_monad_r : Monad U;
      bdist_natural :> Natural (@bdist U T _);
      bdist_join_l :
        `(bdist U T ‚àò join U = map T (join U) ‚àò bdist U T ‚àò map U (bdist U T (A := A)));
      bdist_join_r :
        `(bdist U T ‚àò map U (join T) = join T ‚àò map T (bdist U T) ‚àò bdist U T (A := T A));
      bdist_unit_l :
        `(bdist U T ‚àò ret U (A := T A) = map T (ret U));
      bdist_unit_r :
        `(bdist U T ‚àò map U (ret T) = ret T (A := U A));
    }.

End BeckDistributiveLaw.
