(** This file implements bimonads. A [Bimonad] is a functor equipped
    with a monad/comonad structure and a distributive law over itself,
    which ensures the composition (W ‚àò W) is also a monad and a
    comonad. Additionally there are some coherence laws, below, which
    may be read into several equivalent ways: *)
(**
    - The cojoin is a monad homomorphism (W ‚Üí W ‚àò W) and the counit a homomorphism (W ‚Üí 1)
    - The join is a comonad homomorphism (W ‚àò W ‚Üí W) and the unit a homomorphism (1 ‚Üí W)
    - W is a monoid in the category of comonoids (in the category of endofunctors)
    - W is a comonoid in the category of monoids (in the category of endofunctors)
 *)

(** * Bimonad typeclass *)
(******************************************************************************)
Section Bimonad.

  Context
    (W : Type -> Type)
    `{Map W}
    `{Return W}
    `{Join W}
    `{Cojoin W}
    `{Extract W}
    `{BeckDistributiveLaw W W}.

  Class Bimonad :=
    { bimonad_monad :> Monad W;
      bimonad_comonad :> Comonad W;
      bimonad_distributive_law :> BeckDistributiveLaw W W;
      bimonad_dist_counit_r :
      `(map W (extract W A) ‚àò bdist W W = extract W (W A));
      bimonad_dist_counit_l :
      `(extract W (W A) ‚àò bdist W W = map W (extract W A));
      bimonad_cap :
      `(extract W A ‚àò join W = extract W A ‚àò extract W (W A));
      bimonad_baton :
      `(extract W A ‚àò ret W = @id A);
      bimonad_cup :
      `(cojoin W ‚àò ret W = ret W ‚àò ret W (A := A));
      bimonad_butterfly :
        `(cojoin W ‚àò join W (A := A) =
          map W (join W) ‚àò join W ‚àò map W (bdist W W) ‚àò cojoin W ‚àò map W (cojoin W));
    }.

End Bimonad.

(** * Kleisli lifting operation <<bibind>> *)
(******************************************************************************)
Section Bimonad_kleisli_operations.

  Context
    (W : Type -> Type)
    `{Map W} `{Join W} `{Cojoin W}
    `{BeckDistribution W W}.

  Definition bibind {A B} : (W A -> W B) -> (W A -> W B) :=
    fun f => join W ‚àò map W f ‚àò cojoin W.

  Definition twist : W ‚àò W ‚áí W ‚àò W
    := fun A => map W (join W) ‚àò bdist W W ‚àò map W (cojoin W).

  Definition kcomposebi {A B C} :
    (W B -> W C) ->
    (W A -> W B) ->
    (W A -> W C) :=
    fun g f => join W ‚àò map W g ‚àò twist B ‚àò map W f ‚àò cojoin W.

End Bimonad_kleisli_operations.

Module Notations.
  Notation "g ‚ãÜbi f" := (kcomposebi _ g f) (at level 70) : tealeaves_scope.
End Notations.
