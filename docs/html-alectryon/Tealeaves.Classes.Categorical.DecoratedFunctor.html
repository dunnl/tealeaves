(** * The [shift] operation *)
(* uncurry (incr W) = join (W √ó) *)

Definition shift (F : Type -> Type) `{Map F} `{Monoid_op W} {A} :
  W * F (W * A) -> F (W * A) := map F (uncurry incr) ‚àò strength.

  (E : Type)

  dec : F ‚áí F ‚óã (E √ó).

(** ** Typeclass *)
(******************************************************************************)

  (E : Type)
  (F : Type -> Type)
  `{Map F}
  `{Decorate E F} :=
  { dfun_functor :> Functor F;
    dfun_dec_natural :> Natural (@dec E F _);
    dfun_dec_dec : forall (A : Type),
      dec F (E * A) ‚àò dec F A = map F (cojoin (prod E)) ‚àò dec F A;
    dfun_dec_extract : forall (A : Type),
      map F (extract (E √ó) A) ‚àò dec F A = @id (F A);
  }.

(** ** Decoration-preserving natural transformations *)
(******************************************************************************)
Class DecoratePreservingTransformation
  (F G : Type -> Type)
  `{! Map F} `{Decorate E F}
  `{! Map G} `{Decorate E G}
  (œï : F ‚áí G) :=
  { dectrans_commute : `(œï (E * A) ‚àò dec F A = dec G A ‚àò œï A);
    dectrans_natural : Natural œï;
  }.
