Class Traverse (T : Type -> Type) := traverse :

(** ** "Kleisli" composition *)
(******************************************************************************)
Definition kc2
  {G1 G2 : Type -> Type} `{Map G1} {A B C : Type} :
  (B -> G2 C) ->
  (A -> G1 B) ->
  (A -> (G1 ‚àò G2) C) :=
  fun g f => map (F := G1) (A := B) (B := G2 C) g ‚àò f.

  { trf_traverse_id : forall (A : Type),
      traverse (G := fun A => A) id = @id (T A);
    trf_traverse_traverse :
    forall `{Applicative G1} `{Applicative G2}
      (A B C : Type) (g : B -> G2 C) (f : A -> G1 B),
      map (traverse g) ‚àò traverse f = traverse (T := T) (G := G1 ‚àò G2) (g ‚ãÜ2 f);
    trf_traverse_morphism :
    forall `{morphism : ApplicativeMorphism G1 G2 œï} (A B : Type) (f : A -> G1 B),
      œï (T B) ‚àò traverse f = traverse (œï B ‚àò f);
  }.

(** ** Homomorphisms *)
(******************************************************************************)
Class TraversableMorphism
  (T U : Type -> Type)
  `{Traverse T}
  `{Traverse U}
  (œï : forall (A : Type), T A -> U A) :=
  { trvmon_hom : forall `{Applicative G}
      `(f : A -> G B),
      map (F := G) (œï B) ‚àò @traverse T _ G _ _ _ A B f =
        @traverse U _ G _ _ _ A B f ‚àò œï A;
  }.
