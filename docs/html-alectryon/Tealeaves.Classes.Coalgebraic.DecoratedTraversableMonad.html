    dtm_ret : forall (A B : Type),
      toBatch7 ‚àò ret (T := T) (A := A) = Step (Done (@id (T B))) ‚àò ret (T := (W √ó));

      extract_Batch ‚àò mapfst_Batch (ret ‚àò extract (W := (W √ó))) ‚àò toBatch7 = @id (T A);
